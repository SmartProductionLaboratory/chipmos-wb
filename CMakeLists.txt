cmake_minimum_required(VERSION 3.8)
project(ChipmosWb LANGUAGES CXX)
include_directories(${CMAKE_SOURCE_DIR})
set(CMAKE_BUILD_TYPE Debug)
set(CMAKE_ARCHIVE_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/lib)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR})
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -std=gnu++11")

file(GLOB_RECURSE cppsrc ./src/*.cpp)
add_executable(main main.cpp ${cppsrc})
add_executable(wb7 wb7.cpp ${cppsrc})
add_subdirectory(lib)
# set_property(TARGET main PROPERTY CUDA_SEPARABLE_COMPILATION ON)
target_link_libraries(main lib)
target_link_libraries(wb7 lib)
